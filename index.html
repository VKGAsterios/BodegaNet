Base de datos en tiempo real

URL del Web App
Código para Google Apps Script (versión completa)
// ── BodegaNet Apps Script — leer y escribir ── var SHEETS = ['Config','Cuadrillas','Materiales','Stocks','Kardex','Asignaciones']; function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); var action = data.action; if (action === 'saveAll') { // Guardar estado completo de la base de datos saveSheet(ss, 'Cuadrillas', data.cuadrillas); saveSheet(ss, 'Materiales', data.materiales); saveSheet(ss, 'Stocks', data.stocks); saveSheet(ss, 'Kardex', data.kardex); saveSheet(ss, 'Asignaciones', data.asignaciones); return ok(); } if (action === 'appendRow') { // Agregar una fila a una hoja (para movimientos en tiempo real) var tab = ss.getSheetByName(data.tab); if (!tab) { tab = ss.insertSheet(data.tab); tab.appendRow(data.headers); } else if (tab.getLastRow() === 0) { tab.appendRow(data.headers); } tab.appendRow(data.row); return ok(); } return ok(); } function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var action = e.parameter.action; if (action === 'loadAll') { // Devolver toda la base de datos return ContentService.createTextOutput(JSON.stringify({ ok: true, cuadrillas: readSheet(ss, 'Cuadrillas'), materiales: readSheet(ss, 'Materiales'), stocks: readSheet(ss, 'Stocks'), kardex: readSheet(ss, 'Kardex'), asignaciones:readSheet(ss, 'Asignaciones') })).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ok:true,status:'BodegaNet online'})) .setMimeType(ContentService.MimeType.JSON); } function saveSheet(ss, name, jsonStr) { var tab = ss.getSheetByName(name); if (!tab) tab = ss.insertSheet(name); tab.clearContents(); tab.appendRow(['__json__']); tab.appendRow([jsonStr]); } function readSheet(ss, name) { var tab = ss.getSheetByName(name); if (!tab || tab.getLastRow() < 2) return '[]'; return tab.getRange(2, 1).getValue() || '[]'; } function ok() { return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
⚠ Importante al redesplegar: Cada vez que modifiques el código debes crear una nueva versión en Implementar → Administrar implementaciones → Editar → Nueva versión. Si no, los cambios no se aplican.
